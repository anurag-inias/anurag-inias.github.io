List all k-Colorings

Description

Given an undirected graph and knowing the minimum number of colors needed for its graph coloring, list all permutations.

Note

Graph coloring is an NP-complete problem. So we are not asking the coloring itself. The minimum number of colors needed is already given as part of the input.

// Super simple implementation of an unweighted undirected graph.
// Nodes are expected to be contiguously numbered, starting from 1.
class Graph {
  val neighbours = mutableMapOf<Int, MutableList<Int>>()

  fun connect(first: Int, second: Int): Graph {
    neighbours.putIfAbsent(first, mutableListOf())
    neighbours.putIfAbsent(second, mutableListOf())

    neighbours[first]!! += second
    neighbours[second]!! += first

    return this
  }
}

// Map a color number to a human readable color name.
val Int.color: String
  get() {
    return Color.entries.filter { it.key == this }.map { it.name }.firstOrNull() ?: "$this"
  }

fun allColorings(graph: Graph, maxColors: Int) {
  helper(graph, maxColors,mutableMapOf(), graph.neighbours.keys.toMutableList())
}

fun helper(graph: Graph, maxColors: Int, colors: MutableMap<Int, Int>, uncolored: MutableList<Int>) {
  if (uncolored.isEmpty()) {
    graph.printColors(colors)
    return
  }

  val node = uncolored.removeFirst()

  val neighbourColors = (graph.neighbours[node] ?: listOf())
    .filter { it in colors }
    .map { colors[it] }

  for (c in 1..maxColors) {
    if (c in neighbourColors) continue
    colors[node] = c
    helper(graph, maxColors, colors, uncolored)
    colors.remove(node)
  }

  uncolored.addFirst(node) // backtrack
}
