Computing the parity of a word

Problem

Parity of a word is 1 if number of 1s in the word is odd; otherwise it is 0.

Example

word parity
1011 1
10001000 0

First approach

Hint

x & (x-1) will unset the rightmost set bit.

Solution 
func parity(x int) int {
  p := 0
  for x != 0 {
    p = 1 - p // flip a bit back-and-forth between 0 and 1
    x = x & (x - 1)
  }
  return p
}
Run time

\(O(n)\) where \(n\) is the number of digits.

Second approach

Hint

Lookup table

Solution

Instead of checking individual bit by bit, check parity of N bits at a time.

func parity2(x int) int {
  pmap := map[int]int{
    0b0000: 0,
    0b0001: 1,
    0b0010: 1,
    0b0011: 0,
    0b0100: 1,
    0b0101: 0,
    0b0110: 0,
    0b0111: 1,
    0b1000: 1,
    0b1001: 0,
    0b1010: 0,
    0b1011: 1,
    0b1100: 0,
    0b1101: 1,
    0b1110: 1,
    0b1111: 0,
  }
  mask := 0b1111

  p := 0
  for i := 0; i < 8; i++ {
    if v, ok := pmap[(x & mask)]; ok && v != 0 {
      p = 1 - p
    }
    x = x >> 4
  }

  return p
}
Run time

\(O(n/\omega)\) where \(n\) is the number of digits and \(\omega\) is width of cache keys. For \(\omega = 4\), that's 8 steps for 32-bit integers and 16 steps for 64-bit longs.

// N = 32
// O(log32) = O(5) = O(1)
fun parity(num: Int): Int {
  var p = num
  p = p xor (p ushr 16)
  p = p xor (p ushr 8)
  p = p xor (p ushr 4)
  p = p xor (p ushr 2)
  p = p xor (p ushr 1)
  return p and 1
}

// N = 64
// O(log64) = O(6) = O(1)
fun parity(num: Long): Int {
  var p = num.toInt() // least significant 32 bits of this `Long` value
  p = p xor (num ushr 32).toInt() // most significant 32 bits of this `Long` value
  p = p xor (p ushr 16)
  p = p xor (p ushr 8)
  p = p xor (p ushr 4)
  p = p xor (p ushr 2)
  p = p xor (p ushr 1)
  return p and 1
}
Run time

\(O(\log_{2}n)\) where \(n\) is the number of digits. For 32-bit integers that's 5 steps and for 64-bit longs it's 6 steps.

@Test
fun oneBit() {
  assertThat(parity(0)).isEqualTo(0)
  assertThat(parity(1)).isEqualTo(1)

  assertThat(parity(0L)).isEqualTo(0)
  assertThat(parity(1L)).isEqualTo(1)
}

@Test
fun twoBits() {
  assertThat(parity(0b10)).isEqualTo(1)
  assertThat(parity(0b11)).isEqualTo(0)

  assertThat(parity(0b10L)).isEqualTo(1)
  assertThat(parity(0b11L)).isEqualTo(0)
}

@Test
fun threeBits() {
  assertThat(parity(0b100)).isEqualTo(1)
  assertThat(parity(0b101)).isEqualTo(0)
  assertThat(parity(0b110)).isEqualTo(0)
  assertThat(parity(0b111)).isEqualTo(1)

  assertThat(parity(0b100L)).isEqualTo(1)
  assertThat(parity(0b101L)).isEqualTo(0)
  assertThat(parity(0b110L)).isEqualTo(0)
  assertThat(parity(0b111L)).isEqualTo(1)
}

@Test
fun randomInt() {
  ThreadLocalRandom.current().ints(1000).forEach {
    assertThat(parity(it)).isEqualTo(it.parityValidation())
  }
}

@Test
fun randomLong() {
  ThreadLocalRandom.current().longs(1000).forEach {
    assertThat(parity(it)).isEqualTo(it.parityValidation())
  }
}
