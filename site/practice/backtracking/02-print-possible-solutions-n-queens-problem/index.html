Print all possible solutions to N-Queens problem

Description

We are given a chess board of the size \(N \times N\). Our goal is to place \(N\) queens on this board such that they can't eliminate each other.

Note

A solution exists for all board sizes except \(2\) and \(3\).

Examples

_ Q

Key concept

Process the board left-to-right, top-to-bottom.

and for each cell consider whether a queen can be placed there.

fun placeQueens(boardSize: Int) {
  val board = Array(boardSize) { CharArray(boardSize) { '_' } }
  helper(board, 0)
}

fun helper(board: Array<CharArray>, row: Int) {
  // Reached the end of search space (leaf node).
  if (row == board.size) {
    board.print()
    println()
    return
  }

  for (col in 0..<board.size) {
    if (!board.canPlaceQueen(row, col)) continue
    board[row][col] = 'Q'
    helper(board, row+1)
    board[row][col] = '_' // bracktrack, even on successful paths.
  }
}

fun Array<CharArray>.canPlaceQueen(row: Int, col: Int): Boolean {
  // check diagonal towards top-left corner.
  for ((r, c) in ((row - 1) downTo 0).zip((col-1) downTo 0)) {
    if (this[r][c] == 'Q') return false
  }
  // check diagonal towards top-right corner.
  for ((r, c) in ((row - 1) downTo 0).zip((col+1)..<size)) {
    if (this[r][c] == 'Q') return false
  }
  // check right above this cell.
  for (r in (row - 1) downTo 0) {
    if (this[r][col] == 'Q') return false
  }
  // check left of this cell.
  for (c in (col - 1) downTo 0) {
    if (this[row][c] == 'Q') return false
  }
  return true
}

fun Array<CharArray>.print() {
  for (r in 0..<size) {
    for (c in 0..<size) {
      print("${this[r][c]} ")
    }
    println()
  }
}
