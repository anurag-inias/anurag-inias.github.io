List all Hamiltonian paths

Description

Given an undirected graph, list all Hamiltonian paths present in it. A Hamiltonian path in a path through every vertex visited exactly once. This is similar to Knight's tour problem discussed previously.

// Super simple implementation of an unweighted undirected graph.
// Nodes are expected to be contiguously numbered, starting from 1.
class Graph {
  val neighbours = mutableMapOf<Int, MutableList<Int>>()

  fun connect(first: Int, second: Int): Graph {
    neighbours.putIfAbsent(first, mutableListOf())
    neighbours.putIfAbsent(second, mutableListOf())

    neighbours[first]!! += second
    neighbours[second]!! += first

    return this
  }
}

fun hamiltonPaths(graph: Graph) {
  for (src in graph.neighbours.keys) {
    helper(graph, src, mutableListOf())
  }
}

fun helper(graph: Graph, node: Int, path: MutableList<Int>) {
  path.add(node)
  if (path.size == graph.neighbours.size) {
    println(path)
    path.removeLast()
    return
  }

  val unvisitedNeighbours = (graph.neighbours[node] ?: listOf()).filter { it !in path }
  for (neighbour in unvisitedNeighbours) {
    helper(graph, neighbour, path)
  }

  path.removeLast()
}
