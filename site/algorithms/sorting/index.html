    Selection Sort

    • Split the input in two subarrays. First one is sorted and second one is being sorted.
    • In each iteration, find the smallest element in the second subarray and move it to the front.
    • Shrink the second subarray.

    public void sort(int[] nums) {
  if (nums == null || nums.length < 2) return;

  for (int i = 0; i < nums.length-1; i++) {
    int minIndex = findSmallest(nums, i);
    swap(nums, i, minIndex);
  }
}

private int findSmallest(int[] nums, int start) {
  int minIndex = start;
  for (int i = start+1; i < nums.length; i++)
    if (nums[i] < nums[minIndex])
      minIndex = i;
  return minIndex;
}

    Insertion Sort

    • Split the input in two subarrays. First one is sorted and second one is being sorted.
    • In each iteration, pick one element from the front of second subarray and find its correct placement in first subarray.
    • Shift elements of first subarray to right to make room for processed element.

    • Insertion sort is very similar to selection sort, but is a marked improvement:
    • Selection sort always scans all remaining elements to find the smallest one.
    • whereas insertion sort takes advantages of the sorted property of the first subarray and only scans the minimum needed elements.
    • Insertion sort, however, requires more writes due to excessive swaps. This makes it less preferable for EEPROM / flash storage where writes are expensive.

    public void sort(int[] nums) {
  if (nums == null || nums.length < 2) return;

  for (int i = 1; i < nums.length; i++)
    insert(nums, i, nums[i]);
}

// end exclusive.
private void insert(int[] nums, int end, int value) {
  int i = end-1;
  while (i >= 0 && value < nums[i]) {
    nums[i+1] = nums[i];
    i--;
  }
  nums[i+1] = value;
}

    Merge Sort

    • Builds on divide-and-conquer approach.
    • Each iteration splits input array in two halves and recursively sorts them.
    • Followed by a merge step where two sorted subarrays are places back in original array.

    • Auxiliary space is needed for atleast one half of the input, other merge step will end with non-sense output.
    • Heap sort if often favored when a consist \(O(n \cdot log \ n)\) running time is needed, and quick sort when a generally fast performance is needed.
    • However, merge sort is best choice for linked-list where slow memory access prove detrimental to heapsort and quicksort.

    public void sort(int[] nums) {
  if (nums == null) return;
  sort(nums, 0, nums.length);
}

private void sort(int[] nums, int start, int end) {
  if (start < 0 || end > nums.length || start >= end || start + 1 == end) return;

  if (start + 2 == end) {
    if (nums[start] > nums[start + 1]) swap(nums, start, start + 1);
    return;
  }

  // Move left subarray to auxiliary space.
  int length = end - start;
  int[] left = new int[length / 2];
  System.arraycopy(nums, start, left, 0, left.length);

  // Sort subarrays.
  sort(left, 0, left.length);
  sort(nums, start + left.length, end);

  // merge.
  int l = 0, r = start + left.length, c = start;
  while (l < left.length && r < nums.length) {
    if (left[l] < nums[r]) nums[c++] = left[l++];
    else nums[c++] = nums[r++];
  }
  if (l < left.length) System.arraycopy(left, l, nums, c, left.length - l);
}

    package dev.justanotherblog.sort;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import java.util.Arrays;
import java.util.concurrent.ThreadLocalRandom;

import static org.junit.jupiter.api.Assertions.assertArrayEquals;
import static org.junit.jupiter.api.Assertions.assertNull;

class SortTest {

  private Sort sort;

  @BeforeEach
  void setUp() {
    sort = new MergeSort();
  }

  @Test
  public void testNull() {
    assertNull(sort.getSorted(null));
  }

  @Test
  public void testSingle() {
    assertArrayEquals(new int[]{7}, sort.getSorted(new int[]{7}));
  }

  @Test
  public void testDouble() {
    assertArrayEquals(new int[]{1, 2}, sort.getSorted(new int[]{1, 2}));
    assertArrayEquals(new int[]{1, 2}, sort.getSorted(new int[]{2, 1}));
  }

  @Test
  public void testTriple() {
    assertArrayEquals(new int[]{1, 2, 3}, sort.getSorted(new int[]{1, 2, 3}));
    assertArrayEquals(new int[]{1, 2, 3}, sort.getSorted(new int[]{1, 3, 2}));
    assertArrayEquals(new int[]{1, 2, 3}, sort.getSorted(new int[]{2, 1, 3}));
    assertArrayEquals(new int[]{1, 2, 3}, sort.getSorted(new int[]{2, 3, 1}));
    assertArrayEquals(new int[]{1, 2, 3}, sort.getSorted(new int[]{3, 1, 2}));
    assertArrayEquals(new int[]{1, 2, 3}, sort.getSorted(new int[]{3, 2, 1}));
  }

  @Test
  public void testFuzzy() {
    int[] nums = ThreadLocalRandom.current().ints(100, 0, 80).toArray();
    int[] copy = new int[nums.length];
    System.arraycopy(nums, 0, copy, 0, nums.length);

    Arrays.sort(copy);
    sort.sort(nums);

    assertArrayEquals(copy, nums);
  }
}
