def test_clone():
    l = LinkedList()
    assert copy(l) == l
    assert copy(l) is not l

    l = LinkedList()
    l.prepend(3)
    l.prepend(2)
    l.prepend(1)
    assert copy(l) == l
    assert copy(l) is not l
