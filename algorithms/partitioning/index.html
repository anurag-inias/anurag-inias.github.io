    int partition(int[] array) {
  int pivot = array[0];
  int cursor = 1;

  for (int i = 1; i < array.length; i++)
    if (array[i] < pivot)
      swap(array, cursor++, i);

  swap(array, 0, cursor-1);
  return cursor-1;
}

    package dev.justanotherblog.sort;

import org.junit.jupiter.api.Test;

import java.util.concurrent.ThreadLocalRandom;

import static org.junit.jupiter.api.Assertions.*;

class PartitionTest {

  @Test
  public void testSingle() {
    int[] arr = new int[]{5};
    int pivot = 5;
    int pivotIndex = Partition.partition(arr);

    assertEquals(pivot, arr[pivotIndex]);
    assertLessThanPivot(arr, pivotIndex, pivot);
    assertGreaterThanOrEqualsToPivot(arr, pivotIndex+1, pivot);
  }


  @Test
  public void testDouble() {
    int[] arr = new int[]{5, 4};
    int pivot = 5;
    int pivotIndex = Partition.partition(arr);

    assertEquals(pivot, arr[pivotIndex]);
    assertLessThanPivot(arr, pivotIndex, pivot);
    assertGreaterThanOrEqualsToPivot(arr, pivotIndex+1, pivot);
  }


  @Test
  public void testTriple() {
    int[] arr = new int[]{5, 4, 1};
    int pivot = 5;
    int pivotIndex = Partition.partition(arr);

    assertEquals(pivot, arr[pivotIndex]);
    assertLessThanPivot(arr, pivotIndex, pivot);
    assertGreaterThanOrEqualsToPivot(arr, pivotIndex+1, pivot);
  }

  @Test
  public void testQuadrupole() {
    int[] arr = new int[]{5, 4, 9, 1};
    int pivot = 5;
    int pivotIndex = Partition.partition(arr);

    assertEquals(pivot, arr[pivotIndex]);
    assertLessThanPivot(arr, pivotIndex, pivot);
    assertGreaterThanOrEqualsToPivot(arr, pivotIndex+1, pivot);
  }

  @Test
  public void testFuzzy() {
    int[] arr = ThreadLocalRandom.current().ints(100, 0, 80).toArray();
    int pivot = arr[0];
    int pivotIndex = Partition.partition(arr);

    assertEquals(pivot, arr[pivotIndex]);
    assertLessThanPivot(arr, pivotIndex, pivot);
    assertGreaterThanOrEqualsToPivot(arr, pivotIndex+1, pivot);
  }

  private void assertLessThanPivot(int[] array, int endExclusiveIndex, int pivot) {
    for (int i = 0; i < endExclusiveIndex; i++)
      assertTrue(array[i] < pivot);
  }

  private void assertGreaterThanOrEqualsToPivot(int[] array, int startInclusiveIndex, int pivot) {
    for (int i = startInclusiveIndex; i < array.length; i++)
      assertTrue(array[i] >= pivot);
  }
}

    So what's happening here?

    • We pick the first element (idx = 0) of the array as pivot.
    • Index 0 is omitted from the process.
    • The cursor is where the next element < pivot will be placed.

    When the whole loop is done, we are left with cursor pointing to where the next < pivot element would've gone. At this stage we have:

    • idx = 0 where pivot is,
    • idx = [1, cursor) where elements < pivot are,
    • and idx = [cursor, array.length) where elements ≥ pivot are.

    So a final swap is needed b/w 0 and cursor-1. This splits the array in two parts:

    • \(\big[0, \ cursor\big)\) all \(a_i < pivot\).
    • \(\Big[cursor, \ len(array)\Big)\) all \(a_i \ge pivot\).

    // Half open interval [l, c) where elements equals to pivot are grouped.
public static int[] partition3way(int[] array) {
  int pivot = array[0];
  int l = 0, r = array.length-1, c = 0;

  while (c <= r) {
    switch (Integer.compare(array[c], pivot)) {
      case -1 -> swap(array, l++, c++);
      case 1 -> swap(array, c, r--);
      default -> c++;
    }
  }

  return new int[]{l, c};
}

    package dev.justanotherblog.sort;

import org.junit.jupiter.api.Test;

import java.util.concurrent.ThreadLocalRandom;

import static org.junit.jupiter.api.Assertions.*;

class PartitionTest {

  @Test
  public void testSingle() {
    int[] arr = new int[] {5};
    int pivot = 5;
    int[] pivotIndices = Partition.partition3way(arr);

    assertLessThanPivot(arr, pivotIndices[0], pivot);
    assertEqualsToPivot(arr, pivotIndices[0], pivotIndices[1], pivot);
    assertGreaterThanPivot(arr, pivotIndices[1], pivot);
  }

  @Test
  public void testDouble() {
    int[] arr = new int[] {5, 1};
    int pivot = 5;
    int[] pivotIndices = Partition.partition3way(arr);

    assertLessThanPivot(arr, pivotIndices[0], pivot);
    assertEqualsToPivot(arr, pivotIndices[0], pivotIndices[1], pivot);
    assertGreaterThanPivot(arr, pivotIndices[1], pivot);
  }

  @Test
  public void testTriple() {
    int[] arr = new int[] {5, 1, 9};
    int pivot = 5;
    int[] pivotIndices = Partition.partition3way(arr);

    assertLessThanPivot(arr, pivotIndices[0], pivot);
    assertEqualsToPivot(arr, pivotIndices[0], pivotIndices[1], pivot);
    assertGreaterThanPivot(arr, pivotIndices[1], pivot);
  }

  @Test
  public void testFuzzy() {
    int[] arr = ThreadLocalRandom.current().ints(100, 0, 80).toArray();
    int pivot = arr[0];
    int[] pivotIndices = Partition.partition3way(arr);

    assertLessThanPivot(arr, pivotIndices[0], pivot);
    assertEqualsToPivot(arr, pivotIndices[0], pivotIndices[1], pivot);
    assertGreaterThanPivot(arr, pivotIndices[1], pivot);
  }

  private void assertEqualsToPivot(
      int[] array, int startInclusiveIndex, int endExclusiveIndex, int pivot) {
    for (int i = startInclusiveIndex; i < endExclusiveIndex; i++) assertEquals(pivot, array[i]);
  }

  private void assertLessThanPivot(int[] array, int endExclusiveIndex, int pivot) {
    for (int i = 0; i < endExclusiveIndex; i++) assertTrue(array[i] < pivot);
  }

  private void assertGreaterThanPivot(int[] array, int startInclusiveIndex, int pivot) {
    for (int i = startInclusiveIndex; i < array.length; i++) assertTrue(array[i] > pivot);
  }
}

    Note interesting invariants here:

    • case \(a_c < pivot\) and case \(\text{default}\) ensures that \(c \ge l\) at all times.
    • l always point to an element which is in the middle slice, i.e. == pivot.
    • c is always pointing to the next element which needs to be processed, and in the ends point to the first element of last slice, i.e. > pivot.
